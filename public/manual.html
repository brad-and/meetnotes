function doGet() {
  var cal   = CalendarApp.getDefaultCalendar()
  var now   = new Date()
  var end   = new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000)
  var items = cal.getEvents(now, end)

  var result = items.map(function(e) {
    var guests    = e.getGuestList(true)  // true = 주최자 포함
    var attendees = guests.map(function(g) {
      var status = 'needsAction'
      try {
        var gs = g.getGuestStatus()
        if      (gs === CalendarApp.GuestStatus.YES)   status = 'accepted'
        else if (gs === CalendarApp.GuestStatus.MAYBE) status = 'tentative'
        else if (gs === CalendarApp.GuestStatus.NO)    status = 'declined'
      } catch(err) { /* 일부 시스템 참석자는 status 없음 → needsAction 유지 */ }
      return { email: g.getEmail(), status: status }
    })
    return {
      id:        e.getId().replace('@google.com', ''),
      title:     e.getTitle(),
      start:     e.getStartTime().toISOString(),
      end:       e.getEndTime().toISOString(),
      attendees: attendees,
      location:  e.getLocation() || undefined
    }
  })

  var output = ContentService
    .createTextOutput(JSON.stringify({
      events:    result,
      updatedAt: new Date().toISOString()
    }))
  output.setMimeType(ContentService.MimeType.JSON)
  return output
}

01 · 아이디어회의

왜 만들었나

💡
핵심 문제: 회의가 끝나면 "이 결정 누가 한다고 했더라?" "다음 스텝이 뭐였더라?" — 매번 기억에 의존하거나, 누군가 수작업으로 정리해야 했어요. MeetNotes는 회의를 녹음하기만 하면 AI가 회의록·액션 아이템·다음 스텝을 자동으로 작성하는 서비스입니다.

✨ After

회의 시작 버튼 → 자동 녹음·전사 → 종료 버튼 → 3초 후 AI 회의록 완성. 결정사항·액션 아이템·다음 스텝 자동 분류. Slack 원클릭 전송.

SCREEN 2

녹음·전사 (RecordingScreen)

녹음과 전사가 이루어지는 화면. 대면 회의는 선택한 녹음 방식에 따라 표준 녹음 또는 Deepgram 실시간 대사로 동작하고, 온라인 회의는 파일 업로드로 동작합니다.

  • ▸표준 녹음: 실시간 전사 없이 로컬 녹음 안정성 우선, 종료 후 전체 오디오를 AI가 전사
  • ▸실시간 대사: Deepgram WebSocket으로 녹음 중 발화 표시, 종료 후 대사 기반 회의록 생성
  • ▸온라인: MP3/MP4/WAV/M4A 파일 업로드 후 전사
  • ▸발언자 자동 구분 (Speaker 0, 1, 2...) + 이름 수정 가능

회의 종료 후 전사된 트랜스크립트(또는 오디오 파일)를 Gemini 2.5 Flash에 전달해 구조화된 JSON 회의록을 생성합니다. 할루시네이션 방지를 위해 temperature: 0, systemInstruction 절대원칙 목록, 3단계 사실 기반 구조를 함께 적용합니다. JSON 파싱은 jsonrepair 2차 복구 전략을 사용합니다.

async function postToWebApp(tabId, data) {
  await chrome.scripting.executeScript({
    target: { tabId },
    world: 'MAIN',
    func: (payload) => {
      localStorage.setItem(
        'meetnotes_pending_event',
        JSON.stringify(payload)
      )
      window.dispatchEvent(
        new CustomEvent('meetnotes-from-ext', { detail: payload })
      )
      window.postMessage({ source: 'meetnotes-ext', ...payload }, '*')
    },
    args: [data],
  })
}

③ 익스텐션 팝업 — 녹음 완료 후 상태 불일치 수정

웹앱에서 녹음을 완료해도 익스텐션 팝업이 "녹음 중"으로 남아있던 문제를 수정했습니다.

원인 분석

녹음 종료 시 setStep('review')가 즉시 실행되어 RECORDING_STOPPED 이벤트가 발송되지 않습니다. ANALYSIS_DONE은 전송되지만, background.js의 해당 핸들러에 isRecording: false가 없어 팝업이 계속 "녹음 중"으로 표시됐습니다.

수정 전 (버그)
if (event.type === 'ANALYSIS_DONE') {
  await chrome.storage.session.set({
    // isRecording 없음 → 팝업이 녹음중으로 유지
    isStopping: false, isDone: true,
    result: event.result ?? null,
  })
}
수정 내용

ANALYSIS_DONE 핸들러에 isRecording: false를 추가했습니다. 분석 완료 시점에 반드시 isRecording을 해제해 팝업 상태를 정상화합니다.

수정 후 (extension/background.js)
if (event.type === 'ANALYSIS_DONE') {
  await chrome.storage.session.set({
    // isRecording: false 추가 — RECORDING_STOPPED 생략 보완
    isRecording: false,
    isStopping: false, isDone: true,
    result: event.result ?? null,
    error: event.error ?? null,
  })
}
